# date-range

Returns periods of time in target time range

## Example

```js
let getRanges = require("date-range");

return getRanges("2017-05-21", "2019-06-13");

/*
Returns object

{ 
  day: [ '2017-05-21',
          '2017-05-22',
          '2017-05-23',
          '2017-05-24',
          '2017-05-25',
          '2017-05-26',
          '2017-05-27',
          '2017-05-28',
          '2017-05-29',
          '2017-05-30',
          '2017-05-31',
          '2019-06-01',
          '2019-06-02',
          '2019-06-03',
          '2019-06-04',
          '2019-06-05',
          '2019-06-06',
          '2019-06-07',
          '2019-06-08',
          '2019-06-09',
          '2019-06-10',
          '2019-06-11',
          '2019-06-12',
          '2019-06-13' ],
  month: [ '2017-06', '2019-01', '2019-02', '2019-03', '2019-04', '2019-05' ],
  half_year: [ '2017-2-half-year' ],
  year: [ '2018' ] 
}
*/
```
Also possible to limit maximal granularity


```js

let getRanges = require("date-range");

return getRanges("2017-05-21", "2019-06-13", "half_year");

/*
Returns object

{ day: 
   [ '2017-05-21',

*/
```
